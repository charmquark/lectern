                                                $(containers).lectern(
                                                    'slider',
                                                    {options}
                                                )
                                            
      
                                            
      
                                                new Lectern.Slider(
                                                    $(single container),
                                                    {options}
                                                )
                                            
      
                                        
      
                                        
      
                                            
      Create a new slider. The instance object is always available
                                                through container.data('lectern-slider').
      
                                        
      


                                        
      
                                            $(container).data('lectern-slider')
                                        
      
                                        
      
                                            
      Access an existing slider instance.
      
                                        
      


                                        
      
                                            slider.active
                                        
      
                                        
      
                                            
      Currently active (showing) frame.
      
                                        
      


                                        
      
                                            
      
                                                slider.navigate(frame)
                                            
      
                                            
      
                                                slider.navigate(index)
                                            
      
                                        
      
                                        
      
                                            
      Navigate the slider to the requested frame.
      
                                        
      


                                        
      
                                            
      slider.next()
      
                                            
      slider.prev()
      
                                        
      
                                        
      
                                            
      Programmatically advance a slider to the next or previous frame,
                                            respectively.
      
                                        
      
                                    
      
                                
      • 

                                
    • 
                                    
      
                                        
      
                                            
      
                                                slider.beforeNavigate(handler, data)
                                            
      
                                            
      
                                                slider.afterNavigate(handler, data)

    • 
                                    
      
                                        
      duration 500
      
                                        
      
                                            
      The duration of slide transitions, in milliseconds.
      
                                        
      

                                        
      easing "swing"
      
                                        
      
                                            
      The animation easing prameter.
      
                                        
      

                                        
      indexClasses false
      
                                        
      
                                            
      If enabled, frames will be decorated with classes in the form
                                                frame-N, where N is the frame index.
      
                                        
      

                                        
      ignoreData false
      
                                        
      
                                            
      If enabled, Lectern will not read data attributes to set any options. This is
                                                useful if there is a conflict with another library.
      
                                        
      

                                        
      wrapAround true
      
                                        
      
                                            
      When enabled, navigating past the ends of the slider will wrap around from
                                                the first to last, or last to first, frame.

      classes: {
                                            
      active     'lectern-slider-active'
      
                                            
      canvas     'lectern-slider-canvas'
      
                                            
      caption    'lectern-slider-caption'
      
                                            
      container  'lectern-slider-container'
      
                                            
      frame      'lectern-slider-frame'
      
                                            
      hidden1    'lectern-slider-hidden1'
      
                                            
      hidden2    'lectern-slider-hidden2'
      
                                            
      navigator  'lectern-slider-navigator'
      
                                        }
      
                                        
      
                                            
      Allows you to change what CSS classes will be added
                                                to the various slider parts.
      
                                        
      


                                        
      queries: {
                                            
      canvas     '> ul'
      
                                            
      caption    '> span'
      
                                        }
      
                                        
      
                                            
      If you are trying to be very clever, you can change
                                                the queries Lectern will use to identify what parts
                                                of your markup should become in the final slider.
      
                                        
      


                                        
      states: {
                                            
      active: {
                                                
      opacity    1.0
      
                                                
      left       '0px'
      
                                                
      right      '0px'
      
                                            }
      
                                            
      hidden1: {
                                                
      left   '100%'
      
                                                
      right  '-100%'
      
                                            }
      
                                            
      hidden2: {
                                                
      left   '-100%'
      
                                                
      right  '100%'
      
                                            }
      
                                        }
      
                                        
      
                                            
      These are the various states which will be applied
                                                to slider frames to animate them.  You may modify
                                                them however you need to suit your design.
