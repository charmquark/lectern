                                    $(containers).lectern(
                                        'tabs',
                                        {options}
                                    )
                                
    
                                
    
                                    new Lectern.TabView(
                                        $(single container),
                                        {options}
                                    )
                                
    
                            
    
                            
    
                                
    Create a new TabView. The instance object is always available
                                    through container.data('lectern-tabs') should you
                                    need it.
    
                            
    


                            
    
                                $(container).data('lectern-tabs')
                            
    
                            
    
                                
    Access an existing TabView instance.
    
                            
    


                            
    
                                
    
                                    tabView.navigate(frame)
                                
    
                                
    
                                    tabView.navigate(index)
                                
    
                            
    
                            
    
                                
    Navigate the TabView to the requested frame.
    
                            
    
                        
    
                    
    
                
    • 

                
  • 
                    
    
                        
    
                            
      
                                
    • 
                                    
      
                                        
      duration 200
      
                                        
      
                                            
      The duration of transitions, in milliseconds.
      
                                        
      

                                        
      easing "swing"
      
                                        
      
                                            
      The animation easing prameter.
      
                                        
      

                                        
      height null
      
                                        
      
                                            
      If set, the TabView canvas will be constrained to the
                                                provided height. When not set (the default), the
                                                canvas will have the same height as the tallest
                                                frame.
      
                                        
      

                                        
      indexClasses false
      
                                        
      
                                            
      If enabled, frames will be decorated with classes in
                                                the form frame-N, where N
                                                is the frame index.
      
                                        
      

                                        
      ignoreData false
      
                                        
      
                                            
      If enabled, Lectern will not read data attributes to
                                                set any options. This is useful if there is a
                                                conflict with another library.
      
                                        
      
                                    
      
                                
      • 

                                
    • 
                                    
      
                                        
      classes: {
                                            
      active     'lectern-tabs-active'
      
                                            
      canvas     'lectern-tabs-canvas'
      
                                            
      container  'lectern-tabs-container'
      
                                            
      frame      'lectern-tabs-frame'
      
                                            
      inactive   'lectern-tabs-inactive'
      
                                            
      tab        'lectern-tabs-tab'
      
                                            
      tabBar     'lectern-tabs-tabbar'
      
                                        }
      
                                        
      
                                            
      Allows you to change what CSS classes will be added
                                                to the various TabView parts.
      
                                        
      


                                        
      queries: {
                                            
      canvas '> ul'
      
                                            
      tab    '> .tab'
      
                                        }
      
                                        
      
                                            
      If you are trying to be very clever, you can change
                                                the queries Lectern will use to identify what parts
                                                of your markup should become in the final
                                                TabView.
      

                                            
      Note that the query for tab labels is done in the
                                                context of the frame element, not the container.
      
                                        
      


                                        
      states: {
                                            
      active: {
                                                
      opacity    1
      
                                            }
      
                                            
      inactive: {
                                                
      opacity    0
      
                                            }
      
                                        }
      
                                        
      
                                            
      These are the various states which will be applied
                                                to frames to animate them.  You may modify them
                                                however you need to suit your design.
